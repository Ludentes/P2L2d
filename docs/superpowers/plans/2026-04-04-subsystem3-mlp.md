# CartoonAlive MLP — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the CartoonAlive MLP: a 4-layer network that maps 478 MediaPipe facial landmarks → N Live2D parameter values in real time (<16ms CPU), trained on 100k synthetic renders of the Hiyori rig.

**Architecture:** `RigConfig` (in `rig/config.py`) holds all rig-specific constants; the MLP takes `len(rig_config.param_ids)` as output size at construction time. Training data is generated by rendering the Hiyori `.moc3` via `live2d-py` + pygame offscreen, running MediaPipe FaceMesh on each frame, and saving `(landmarks[956], params[N])` pairs. Normalisation statistics are computed from the dataset and baked as frozen buffers into the model before export.

**Tech Stack:** Python 3.12, uv, PyTorch ≥2.3, live2d-py 0.6.1.1, mediapipe ≥0.10, pygame ≥2.5, numpy ≥1.26, Pillow ≥10.0

```python
import numpy as np
import torch
import torch.nn as nn


class InputNorm(nn.Module):
    """Non-trainable input normalisation. Subtracts mean, divides by std.
    Stats are set after dataset generation via set_stats() and frozen."""

    def __init__(self, in_features: int) -> None:
        super().__init__()
        self.register_buffer("mean", torch.zeros(in_features))
        self.register_buffer("std", torch.ones(in_features))

    def set_stats(self, mean: np.ndarray, std: np.ndarray) -> None:
        self.mean.copy_(torch.from_numpy(mean).float())
        self.std.copy_(torch.from_numpy(std).float().clamp(min=1e-6))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return (x - self.mean) / self.std


class OutputDenorm(nn.Module):
    """Non-trainable output denormalisation. Multiplies by std, adds mean.
    Inverts the normalisation applied to training targets."""

    def __init__(self, out_features: int) -> None:
        super().__init__()
        self.register_buffer("mean", torch.zeros(out_features))
        self.register_buffer("std", torch.ones(out_features))

    def set_stats(self, mean: np.ndarray, std: np.ndarray) -> None:
        self.mean.copy_(torch.from_numpy(mean).float())
        self.std.copy_(torch.from_numpy(std).float().clamp(min=1e-6))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return x * self.std + self.mean


class CartoonAliveMLP(nn.Module):
    """4-layer MLP with LayerNorm, GELU, and one residual skip at the 512→256 level.

    Architecture:
        InputNorm(956)
        Linear(956, 512) → LayerNorm(512) → GELU
        Linear(512, 256) → LayerNorm(256) → GELU  +  skip(Linear(512, 256))
        Linear(256, 128) → LayerNorm(128) → GELU
        Linear(128, N)
        OutputDenorm(N)
    """

    def __init__(self, n_params: int) -> None:
        super().__init__()
        self.input_norm = InputNorm(956)

        self.fc1 = nn.Linear(956, 512)
        self.ln1 = nn.LayerNorm(512)

        self.fc2 = nn.Linear(512, 256)
        self.ln2 = nn.LayerNorm(256)
        self.skip = nn.Linear(512, 256)  # residual projection

        self.fc3 = nn.Linear(256, 128)
        self.ln3 = nn.LayerNorm(128)

        self.fc4 = nn.Linear(128, n_params)
        self.output_denorm = OutputDenorm(n_params)

        self.act = nn.GELU()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = self.input_norm(x)

        h1 = self.act(self.ln1(self.fc1(x)))
        h2 = self.act(self.ln2(self.fc2(h1)) + self.skip(h1))
        h3 = self.act(self.ln3(self.fc3(h2)))
        out = self.fc4(h3)

        return self.output_denorm(out)

    def set_norm_stats(
        self,
        input_mean: np.ndarray,
        input_std: np.ndarray,
        output_mean: np.ndarray,
        output_std: np.ndarray,
    ) -> None:
        """Bake normalisation statistics into the model as frozen buffers."""
        self.input_norm.set_stats(input_mean, input_std)
        self.output_denorm.set_stats(output_mean, output_std)
```

```python
"""Training data generation for the CartoonAlive MLP.

Stratified sampling strategy (100k total):
  - 20% base poses: 8 canonical expressions × 2,500, each with σ=3 Gaussian perturbation
  - 50% natural random: Normal(0, 8) clipped to PARAM_RANGE
  - 20% extreme: uniform over full PARAM_RANGE
  - 10% symmetric pairs: left/right bilateral params are mirrored

Rendering (live2d-py + MediaPipe) is handled by generate_dataset(); call that to
produce a full dataset. The sampling functions in this module are pure-Python and
require no special dependencies.
"""
from __future__ import annotations

import numpy as np
from pathlib import Path
from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from rig.config import RigConfig

# All params use this range (CartoonAlive paper: ±30)
PARAM_RANGE: tuple[float, float] = (-30.0, 30.0)

# Bilateral param pairs: (left_suffix, right_suffix). Used for symmetric sampling.
_BILATERAL_SUFFIXES: list[tuple[str, str]] = [
    ("EyeLOpen", "EyeROpen"),
    ("EyeLSmile", "EyeRSmile"),
    ("BrowLY", "BrowRY"),
    ("BrowLX", "BrowRX"),
    ("BrowLAngle", "BrowRAngle"),
    ("BrowLForm", "BrowRForm"),
    ("ArmLA", "ArmRA"),
    ("ArmLB", "ArmRB"),
    ("HandLB", "HandRB"),
    ("HandL", "HandR"),
]

# 8 canonical expressions: {param_id: value} (unspecified params default to 0.0)
BASE_POSES: dict[str, dict[str, float]] = {
    "neutral": {},
    "smile": {
        "ParamMouthForm": 30.0,
        "ParamEyeLSmile": 30.0,
        "ParamEyeRSmile": 30.0,
        "ParamCheek": 30.0,
    },
    "surprised": {
        "ParamEyeLOpen": 30.0,
        "ParamEyeROpen": 30.0,
        "ParamMouthOpenY": 30.0,
        "ParamBrowLY": 20.0,
        "ParamBrowRY": 20.0,
    },
    "sad": {
        "ParamBrowLAngle": -20.0,
        "ParamBrowRAngle": -20.0,
        "ParamBrowLForm": -15.0,
        "ParamBrowRForm": -15.0,
        "ParamMouthForm": -20.0,
    },
    "blink_l": {
        "ParamEyeLOpen": -30.0,
    },
    "blink_r": {
        "ParamEyeROpen": -30.0,
    },
    "mouth_open": {
        "ParamMouthOpenY": 30.0,
    },
    "head_left": {
        "ParamAngleX": -30.0,
        "ParamBodyAngleX": -15.0,
    },
}


def _pose_to_array(pose: dict[str, float], param_ids: list[str]) -> np.ndarray:
    """Convert a {param_id: value} dict to an ordered array matching param_ids."""
    arr = np.zeros(len(param_ids), dtype=np.float32)
    for i, pid in enumerate(param_ids):
        if pid in pose:
            arr[i] = pose[pid]
    return arr


def sample_params_base_poses(
    rig: "RigConfig",
    n_samples: int,
    rng: np.random.Generator,
) -> np.ndarray:
    """Sample n_samples from the 8 base poses with σ=3 Gaussian perturbation."""
    pose_names = list(BASE_POSES.keys())
    n_per_pose = n_samples // len(pose_names)
    remainder = n_samples - n_per_pose * len(pose_names)

    rows: list[np.ndarray] = []
    for i, name in enumerate(pose_names):
        n = n_per_pose + (1 if i < remainder else 0)
        base = _pose_to_array(BASE_POSES[name], rig.param_ids)
        noise = rng.normal(0.0, 3.0, size=(n, len(rig.param_ids))).astype(np.float32)
        samples = np.clip(base + noise, PARAM_RANGE[0], PARAM_RANGE[1])
        rows.append(samples)

    return np.concatenate(rows, axis=0)


def sample_params_natural(
    rig: "RigConfig",
    n_samples: int,
    rng: np.random.Generator,
) -> np.ndarray:
    """Sample from Normal(0, 8) clipped to PARAM_RANGE — centred on neutral."""
    raw = rng.normal(0.0, 8.0, size=(n_samples, len(rig.param_ids))).astype(np.float32)
    return np.clip(raw, PARAM_RANGE[0], PARAM_RANGE[1])


def sample_params_extreme(
    rig: "RigConfig",
    n_samples: int,
    rng: np.random.Generator,
) -> np.ndarray:
    """Sample uniformly over full PARAM_RANGE."""
    return rng.uniform(
        PARAM_RANGE[0], PARAM_RANGE[1], size=(n_samples, len(rig.param_ids))
    ).astype(np.float32)


def sample_params_symmetric_pairs(
    rig: "RigConfig",
    n_samples: int,
    rng: np.random.Generator,
) -> np.ndarray:
    """Sample random params then mirror bilateral left/right pairs."""
    samples = sample_params_natural(rig, n_samples, rng)

    # Build an index map for bilateral pairs that exist in this rig's param_ids
    id_to_idx = {pid: i for i, pid in enumerate(rig.param_ids)}
    for left_suffix, right_suffix in _BILATERAL_SUFFIXES:
        left_key = f"Param{left_suffix}"
        right_key = f"Param{right_suffix}"
        if left_key in id_to_idx and right_key in id_to_idx:
            li, ri = id_to_idx[left_key], id_to_idx[right_key]
            # Mirror: both sides get the left value
            samples[:, ri] = samples[:, li]

    return samples


def sample_params(
    rig: "RigConfig",
    n_samples: int,
    rng: np.random.Generator,
) -> np.ndarray:
    """Generate n_samples parameter vectors using the stratified sampling strategy.

    Strata:
        20% base poses, 50% natural random, 20% extreme, 10% symmetric pairs
    """
    n_base = round(n_samples * 0.20)
    n_natural = round(n_samples * 0.50)
    n_extreme = round(n_samples * 0.20)
    n_sym = n_samples - n_base - n_natural - n_extreme  # remainder = 10%

    parts = [
        sample_params_base_poses(rig, n_base, rng),
        sample_params_natural(rig, n_natural, rng),
        sample_params_extreme(rig, n_extreme, rng),
        sample_params_symmetric_pairs(rig, n_sym, rng),
    ]
    combined = np.concatenate(parts, axis=0)
    # Shuffle so strata are interleaved
    idx = rng.permutation(len(combined))
    return combined[idx]
```

```python
# ─── Rendering pipeline (requires live2d-py, pygame, MediaPipe) ──────────────

import os
import logging
from contextlib import contextmanager
from typing import Iterator

logger = logging.getLogger(__name__)

# 20 pastel background colours (R, G, B) — prevents MediaPipe confusion on white
_BG_COLORS: list[tuple[int, int, int]] = [
    (220, 230, 240), (240, 220, 230), (230, 240, 220),
    (240, 235, 215), (215, 235, 240), (235, 215, 240),
    (250, 245, 230), (230, 250, 245), (245, 230, 250),
    (240, 248, 255), (255, 240, 245), (245, 255, 240),
    (255, 253, 208), (208, 255, 253), (253, 208, 255),
    (230, 230, 250), (250, 230, 230), (230, 250, 230),
    (255, 228, 225), (225, 255, 228),
]


def _init_mediapipe():
    import mediapipe as mp
    face_mesh = mp.solutions.face_mesh.FaceMesh(
        static_image_mode=True,
        max_num_faces=1,
        refine_landmarks=True,
        min_detection_confidence=0.5,
    )
    return face_mesh


class LiveRenderer:
    """Context manager that owns the pygame + live2d-py OpenGL context.

    Usage:
        with LiveRenderer(rig_config, size=512) as renderer:
            frame = renderer.render(params)  # returns (H, W, 4) uint8 RGBA
    """

    def __init__(self, rig: "RigConfig", size: int = 512) -> None:
        self._rig = rig
        self._size = size
        self._model = None

    def __enter__(self) -> "LiveRenderer":
        os.environ.setdefault("SDL_VIDEODRIVER", "offscreen")
        import pygame
        from pygame.locals import DOUBLEBUF, OPENGL
        import live2d.v3 as live2d

        pygame.init()
        pygame.display.set_mode(
            (self._size, self._size), DOUBLEBUF | OPENGL
        )
        live2d.init()

        self._live2d = live2d
        self._model = live2d.LAppModel()
        self._model.LoadAssets(str(self._rig.model3_json_path))
        return self

    def __exit__(self, *_: object) -> None:
        import pygame
        if self._model is not None:
            self._live2d.dispose()
        pygame.quit()

    def render(self, params: np.ndarray) -> np.ndarray:
        """Apply param vector and render one frame. Returns (H, W, 4) uint8 RGBA."""
        from OpenGL.GL import (
            glReadPixels, GL_RGBA, GL_UNSIGNED_BYTE,
        )

        for i, pid in enumerate(self._rig.param_ids):
            self._model.SetParameterValue(pid, float(params[i]), 1.0)

        self._model.Update()
        self._live2d.clearBuffer(0.0, 0.0, 0.0, 0.0)
        self._model.Draw()

        raw = glReadPixels(0, 0, self._size, self._size, GL_RGBA, GL_UNSIGNED_BYTE)
        frame = np.frombuffer(raw, dtype=np.uint8).reshape(self._size, self._size, 4)
        return np.flipud(frame).copy()  # OpenGL is bottom-up


def _composite_on_bg(
    rgba: np.ndarray, bg: tuple[int, int, int]
) -> np.ndarray:
    """Alpha-composite RGBA frame onto a solid background colour.
    Returns (H, W, 3) uint8 RGB."""
    alpha = rgba[:, :, 3:4].astype(np.float32) / 255.0
    fg = rgba[:, :, :3].astype(np.float32)
    bg_arr = np.array(bg, dtype=np.float32)
    composited = (fg * alpha + bg_arr * (1 - alpha)).clip(0, 255).astype(np.uint8)
    return composited


def _augment(
    img: np.ndarray, rng: np.random.Generator
) -> np.ndarray:
    """Apply mild augmentation: Gaussian blur, JPEG compression, brightness jitter.
    Input/output: (H, W, 3) uint8 RGB numpy array."""
    from PIL import Image, ImageFilter, ImageEnhance
    import io

    pil = Image.fromarray(img)

    # Gaussian blur σ ∈ [0, 0.8]
    sigma = rng.uniform(0.0, 0.8)
    if sigma > 0.1:
        pil = pil.filter(ImageFilter.GaussianBlur(radius=sigma))

    # JPEG compression quality ∈ [85, 100]
    quality = int(rng.integers(85, 101))
    buf = io.BytesIO()
    pil.save(buf, format="JPEG", quality=quality)
    buf.seek(0)
    pil = Image.open(buf).convert("RGB")

    # Brightness jitter ±10%
    factor = rng.uniform(0.9, 1.1)
    pil = ImageEnhance.Brightness(pil).enhance(factor)

    return np.array(pil)


def render_and_detect(
    renderer: LiveRenderer,
    params: np.ndarray,
    bg: tuple[int, int, int],
    rng: np.random.Generator,
    face_mesh=None,
) -> np.ndarray | None:
    """Render rig with params, composite on bg, augment, run MediaPipe.

    Returns:
        ndarray of shape (956,) with [x0,y0, x1,y1, ...] landmark coords
        in normalised image space [0,1], or None if detection failed.
    """
    import mediapipe as mp

    rgba = renderer.render(params)
    rgb = _composite_on_bg(rgba, bg)
    rgb_aug = _augment(rgb, rng)

    if face_mesh is None:
        face_mesh = _init_mediapipe()

    results = face_mesh.process(rgb_aug)

    if not results.multi_face_landmarks:
        return None

    lms = results.multi_face_landmarks[0].landmark
    if len(lms) < 468:
        return None

    # Use first 478 landmarks (refine_landmarks=True gives 478)
    coords = np.array(
        [[lm.x, lm.y] for lm in lms[:478]], dtype=np.float32
    ).flatten()  # shape (956,)
    return coords


def generate_dataset(
    rig: "RigConfig",
    n_samples: int,
    output_dir: Path,
    seed: int = 42,
) -> None:
    """Generate n_samples (landmarks, params) pairs and save to output_dir.

    Files written:
        output_dir/landmarks.npy  — float32 (N, 956)
        output_dir/params.npy     — float32 (N, len(rig.param_ids))
        output_dir/discard_log.txt — discard rate per 1000 attempts
    """
    output_dir = Path(output_dir)
    output_dir.mkdir(parents=True, exist_ok=True)

    rng = np.random.default_rng(seed)
    all_params = sample_params(rig, n_samples, rng)

    landmarks_list: list[np.ndarray] = []
    params_list: list[np.ndarray] = []
    discarded = 0
    resample_queue = list(range(n_samples))

    with LiveRenderer(rig, size=512) as renderer:
        face_mesh = _init_mediapipe()
        bg_colors = _BG_COLORS

        collected = 0
        attempt = 0
        while collected < n_samples:
            if attempt < len(all_params):
                param_vec = all_params[attempt]
            else:
                # Resample: generate a fresh natural random param
                param_vec = sample_params_natural(rig, 1, rng)[0]

            bg = bg_colors[rng.integers(len(bg_colors))]
            lm = render_and_detect(renderer, param_vec, bg, rng, face_mesh)

            if lm is None:
                discarded += 1
                logger.debug("Sample %d discarded (detection failed)", attempt)
            else:
                landmarks_list.append(lm)
                params_list.append(param_vec)
                collected += 1

            attempt += 1
            if attempt % 1000 == 0:
                rate = discarded / attempt
                logger.info(
                    "Progress: %d/%d collected, discard rate %.1f%%",
                    collected, n_samples, rate * 100,
                )

    np.save(output_dir / "landmarks.npy", np.stack(landmarks_list))
    np.save(output_dir / "params.npy", np.stack(params_list))

    with open(output_dir / "discard_log.txt", "w") as f:
        f.write(f"total_attempts={attempt}\n")
        f.write(f"discarded={discarded}\n")
        f.write(f"discard_rate={discarded/attempt:.4f}\n")

    logger.info(
        "Dataset saved: %d samples, discard rate %.1f%%",
        n_samples, discarded / attempt * 100,
    )
```

```python
"""CartoonAlive MLP training loop.

Usage:
    uv run python -m mlp.train \
        --data-dir mlp/data/hiyori \
        --output-dir mlp/checkpoints/hiyori \
        --n-params 74

The data directory must contain landmarks.npy (N, 956) and params.npy (N, n_params).
"""
from __future__ import annotations

import logging
from pathlib import Path

import numpy as np
import torch
import torch.nn as nn
from torch.utils.data import DataLoader, TensorDataset, random_split

from mlp.model import CartoonAliveMLP

logger = logging.getLogger(__name__)


def _load_dataset(data_dir: Path) -> tuple[np.ndarray, np.ndarray]:
    landmarks = np.load(data_dir / "landmarks.npy")
    params = np.load(data_dir / "params.npy")
    assert landmarks.shape[0] == params.shape[0], "landmarks/params row count mismatch"
    assert landmarks.shape[1] == 956, f"Expected 956 landmark features, got {landmarks.shape[1]}"
    return landmarks, params


def train(
    data_dir: Path,
    output_dir: Path,
    n_params: int,
    epochs: int = 100,
    batch_size: int = 512,
    lr: float = 3e-4,
    weight_decay: float = 1e-4,
    val_fraction: float = 0.1,
    early_stop_patience: int = 10,
) -> None:
    """Train CartoonAliveMLP, bake normalisation stats, save model.

    Saved files:
        output_dir/model.pt      — state dict with baked InputNorm/OutputDenorm
        output_dir/val_rmse.txt  — one RMSE line per epoch
    """
    data_dir = Path(data_dir)
    output_dir = Path(output_dir)
    output_dir.mkdir(parents=True, exist_ok=True)

    landmarks, params = _load_dataset(data_dir)

    # Compute and bake normalisation stats
    lm_mean = landmarks.mean(axis=0)
    lm_std = landmarks.std(axis=0)
    p_mean = params.mean(axis=0)
    p_std = params.std(axis=0)

    # Normalise targets for training loss
    params_norm = (params - p_mean) / np.clip(p_std, 1e-6, None)

    X = torch.from_numpy(landmarks)
    Y = torch.from_numpy(params_norm.astype(np.float32))

    dataset = TensorDataset(X, Y)
    n_val = max(1, int(len(dataset) * val_fraction))
    n_train = len(dataset) - n_val
    train_set, val_set = random_split(
        dataset, [n_train, n_val],
        generator=torch.Generator().manual_seed(42),
    )

    train_loader = DataLoader(train_set, batch_size=batch_size, shuffle=True)
    val_loader = DataLoader(val_set, batch_size=batch_size)

    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    logger.info("Training on %s", device)

    model = CartoonAliveMLP(n_params=n_params).to(device)
    model.set_norm_stats(lm_mean, lm_std, p_mean, p_std)

    optimizer = torch.optim.AdamW(
        model.parameters(), lr=lr, weight_decay=weight_decay
    )
    scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=epochs)
    criterion = nn.MSELoss()

    best_val_rmse = float("inf")
    patience_counter = 0
    rmse_log: list[str] = []

    for epoch in range(1, epochs + 1):
        model.train()
        for xb, yb in train_loader:
            xb, yb = xb.to(device), yb.to(device)
            # Forward pass uses InputNorm internally; we pass raw landmarks
            # but train against normalised targets
            # InputNorm normalises x; OutputDenorm denormalises output
            # For training MSE we compare model output (denormed) vs original params
            # To get MSE in normalised space: disable OutputDenorm during training
            pred_norm = model.fc4(
                model.act(model.ln3(model.fc3(
                    model.act(model.ln2(model.fc2(
                        model.act(model.ln1(model.fc1(
                            model.input_norm(xb)
                        )))
                    )) + model.skip(
                        model.act(model.ln1(model.fc1(
                            model.input_norm(xb)
                        )))
                    ))
                )))
            )
            loss = criterion(pred_norm, yb)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

        scheduler.step()

        model.eval()
        val_sq_errors: list[float] = []
        with torch.no_grad():
            for xb, yb in val_loader:
                xb, yb = xb.to(device), yb.to(device)
                pred_norm = model.fc4(
                    model.act(model.ln3(model.fc3(
                        model.act(model.ln2(model.fc2(
                            model.act(model.ln1(model.fc1(
                                model.input_norm(xb)
                            )))
                        )) + model.skip(
                            model.act(model.ln1(model.fc1(
                                model.input_norm(xb)
                            )))
                        ))
                    )))
                )
                sq = (pred_norm - yb).pow(2).mean(dim=1)
                val_sq_errors.extend(sq.cpu().tolist())
        val_rmse = float(np.sqrt(np.mean(val_sq_errors)))
        rmse_log.append(f"{epoch},{val_rmse:.6f}")
        logger.info("Epoch %d/%d  val_rmse=%.4f", epoch, epochs, val_rmse)

        if val_rmse < best_val_rmse:
            best_val_rmse = val_rmse
            patience_counter = 0
            torch.save(model.state_dict(), output_dir / "model.pt")
        else:
            patience_counter += 1
            if patience_counter >= early_stop_patience:
                logger.info("Early stopping at epoch %d", epoch)
                break

    (output_dir / "val_rmse.txt").write_text("\n".join(rmse_log) + "\n")
    logger.info("Training complete. Best val RMSE: %.4f", best_val_rmse)


if __name__ == "__main__":
    import argparse

    logging.basicConfig(level=logging.INFO)
    parser = argparse.ArgumentParser()
    parser.add_argument("--data-dir", type=Path, required=True)
    parser.add_argument("--output-dir", type=Path, required=True)
    parser.add_argument("--n-params", type=int, required=True)
    parser.add_argument("--epochs", type=int, default=100)
    parser.add_argument("--batch-size", type=int, default=512)
    parser.add_argument("--lr", type=float, default=3e-4)
    args = parser.parse_args()

    train(
        data_dir=args.data_dir,
        output_dir=args.output_dir,
        n_params=args.n_params,
        epochs=args.epochs,
        batch_size=args.batch_size,
        lr=args.lr,
    )
```

```python
"""Runtime MLP inference for the CartoonAlive face-to-parameter bridge.

Usage:
    from mlp.infer import MLPInference
    infer = MLPInference(model_path=Path("mlp/checkpoints/hiyori/model.pt"), n_params=74)
    params = infer.predict(landmarks_956)   # np.ndarray (956,) → (74,)
"""
from __future__ import annotations

import logging
import time
from pathlib import Path

import numpy as np
import torch

from mlp.model import CartoonAliveMLP

logger = logging.getLogger(__name__)

_LATENCY_BUDGET_MS = 16.0


class MLPInference:
    """Loads a trained CartoonAliveMLP and exposes predict() for runtime use.

    The model runs on CPU; GPU is reserved for ComfyUI.
    """

    def __init__(self, model_path: Path, n_params: int) -> None:
        self._model = CartoonAliveMLP(n_params=n_params)
        self._model.load_state_dict(
            torch.load(model_path, map_location="cpu", weights_only=True)
        )
        self._model.eval()
        self._n_params = n_params

        median_ms = self.benchmark(n_calls=50)
        if median_ms > _LATENCY_BUDGET_MS:
            logger.warning(
                "MLP inference median %.1fms exceeds %.0fms budget — "
                "runtime bridge should fall back to direct blendshape mapping",
                median_ms,
                _LATENCY_BUDGET_MS,
            )
        else:
            logger.info("MLP inference median %.2fms (budget: %.0fms)", median_ms, _LATENCY_BUDGET_MS)

    def predict(self, landmarks: np.ndarray) -> np.ndarray:
        """Map landmarks → Live2D parameters.

        Args:
            landmarks: float32 array of shape (956,) or (B, 956)

        Returns:
            float32 array of shape (n_params,) or (B, n_params)
        """
        single = landmarks.ndim == 1
        if landmarks.ndim == 1:
            landmarks = landmarks[np.newaxis]

        if landmarks.shape[-1] != 956:
            raise ValueError(
                f"Expected 956 landmark features, got {landmarks.shape[-1]}"
            )

        x = torch.from_numpy(landmarks.astype(np.float32))
        with torch.no_grad():
            out = self._model(x)

        result = out.numpy()
        return result[0] if single else result

    def benchmark(self, n_calls: int = 50) -> float:
        """Run n_calls forward passes on CPU and return median latency in ms."""
        x = torch.zeros(1, 956)
        latencies: list[float] = []
        for _ in range(n_calls):
            t0 = time.perf_counter()
            with torch.no_grad():
                self._model(x)
            latencies.append((time.perf_counter() - t0) * 1000)
        return float(np.median(latencies))
```
