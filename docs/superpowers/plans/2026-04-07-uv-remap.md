> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Remap Hiyori's moc3 UV coordinates so each semantic region gets a non-overlapping rectangular atlas area, enabling clean per-region AI texture editing.

**Architecture:** Per-region bounding-box packing. Group art meshes by semantic part, compute UV bboxes per region, shelf-pack into a single 2048px atlas, transform UVs with linear translate+scale, copy pixel rectangles to new atlas. All mesh topology/deformers/parameters unchanged — only `uv.xys` and `art_mesh.texture_indices` are modified.

**Tech Stack:** Python 3.12, pipeline/moc3.py (moc3 reader/writer), numpy, Pillow, cv2 (for visualization only)

"""UV remap — rearrange moc3 UV layout into clean rectangular regions for AI editing."""
from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path

import numpy as np


@dataclass
class RegionDef:
    """Defines a semantic region as a group of Live2D parts."""
    name: str
    part_names: list[str]


@dataclass
class RegionBBox:
    """A region's UV bounding box on a specific texture sheet."""
    name: str
    texture_index: int
    min_u: float
    min_v: float
    max_u: float
    max_v: float
    mesh_indices: list[int] = field(default_factory=list)

    @property
    def width(self) -> float:
        return self.max_u - self.min_u

    @property
    def height(self) -> float:
        return self.max_v - self.min_v


@dataclass
class PackedRegion:
    """A region assigned to a position on the new atlas."""
    name: str
    old_bbox: RegionBBox
    new_x: int  # pixel x on new atlas
    new_y: int  # pixel y on new atlas
    new_w: int  # pixel width on new atlas
    new_h: int  # pixel height on new atlas


def define_regions() -> list[RegionDef]:
    """Default Hiyori region groupings based on part hierarchy."""
    return [
        RegionDef("face", ["PartFace"]),

from pipeline.moc3 import Moc3


def compute_region_bboxes(moc: Moc3, regions: list[RegionDef]) -> list[RegionBBox]:
    """Compute per-region UV bounding boxes from mesh data.

    Groups art meshes by parent part → region mapping.
    Returns one RegionBBox per region (skips regions with no meshes).
    """
    part_ids = moc["part.ids"]
    mesh_parent_parts = moc["art_mesh.parent_part_indices"]
    mesh_tex_indices = moc["art_mesh.texture_indices"]
    uv_begins = moc["art_mesh.uv_begin_indices"]
    vertex_counts = moc["art_mesh.vertex_counts"]
    all_uvs = moc["uv.xys"]

    # Build part_name → region_name lookup
    part_to_region: dict[str, str] = {}
    for region in regions:
        for pname in region.part_names:
            part_to_region[pname] = region.name

    # Group meshes by region
    region_meshes: dict[str, list[int]] = {r.name: [] for r in regions}
    for mesh_idx in range(len(mesh_parent_parts)):
        part_idx = mesh_parent_parts[mesh_idx]
        if part_idx < 0 or part_idx >= len(part_ids):
            continue
        part_name = part_ids[part_idx]
        rname = part_to_region.get(part_name)
        if rname is not None:
            region_meshes[rname].append(mesh_idx)

    # Compute bboxes
    result: list[RegionBBox] = []
    for region in regions:
        indices = region_meshes[region.name]
        if not indices:
            # Empty region — create zero-size bbox
            result.append(RegionBBox(region.name, 0, 0.0, 0.0, 0.0, 0.0, []))
            continue

        us: list[float] = []
        vs: list[float] = []
        tex_idx = mesh_tex_indices[indices[0]]

        for mi in indices:
            vc = vertex_counts[mi]
            if vc == 0:
                continue
            uv_start = uv_begins[mi] * 2  # flat x,y pairs
            for j in range(vc):
                us.append(all_uvs[uv_start + j * 2])
                vs.append(all_uvs[uv_start + j * 2 + 1])

        if not us:
            result.append(RegionBBox(region.name, tex_idx, 0.0, 0.0, 0.0, 0.0, indices))
            continue

        result.append(RegionBBox(
            name=region.name,
            texture_index=tex_idx,
            min_u=min(us),
            min_v=min(vs),
            max_u=max(us),
            max_v=max(vs),
            mesh_indices=indices,
        ))

    return result
```

def pack_regions(
    bboxes: list[RegionBBox],
    atlas_size: int = 2048,
    padding: int = 4,
) -> list[PackedRegion]:
    """Shelf-pack region bboxes into a new atlas layout.

    Sorts by height (tallest first), places left-to-right on shelves.
    Skips empty regions (zero width or height).
    All regions go onto a single texture sheet.
    """
    # Convert UV bboxes to pixel dimensions on original texture
    sized = []
    for b in bboxes:
        if b.width <= 0 or b.height <= 0 or not b.mesh_indices:
            continue
        # Original pixel size (on original atlas)
        orig_tex_size = atlas_size  # assume square atlas
        pw = int(np.ceil(b.width * orig_tex_size)) + padding
        ph = int(np.ceil(b.height * orig_tex_size)) + padding
        sized.append((b, pw, ph))

    # Sort by height descending (shelf packing heuristic)
    sized.sort(key=lambda x: x[2], reverse=True)

    # Shelf packing
    result: list[PackedRegion] = []
    shelf_y = 0
    shelf_h = 0
    cursor_x = 0

    for bbox, pw, ph in sized:
        if cursor_x + pw > atlas_size:
            # Start new shelf
            shelf_y += shelf_h
            shelf_h = 0
            cursor_x = 0

        if shelf_y + ph > atlas_size:
            raise ValueError(
                f"Region '{bbox.name}' doesn't fit in {atlas_size}x{atlas_size} atlas. "
                f"Need ({cursor_x + pw}, {shelf_y + ph}). Try atlas_size=4096."
            )

        result.append(PackedRegion(
            name=bbox.name,
            old_bbox=bbox,
            new_x=cursor_x,
            new_y=shelf_y,
            new_w=pw - padding,  # strip padding from the stored size
            new_h=ph - padding,
        ))

        cursor_x += pw
        shelf_h = max(shelf_h, ph)

    return result
```

def remap_uvs(moc: Moc3, packing: list[PackedRegion], atlas_size: int = 2048) -> None:
    """Modify moc3 UV coordinates in-place to match new atlas layout.

    For each packed region, transforms all vertex UVs from old bbox position
    to new atlas position. Also sets all remapped mesh texture_indices to 0.
    """
    uv_begins = moc["art_mesh.uv_begin_indices"]
    vertex_counts = moc["art_mesh.vertex_counts"]
    uvs = moc["uv.xys"]  # flat list of f32, (x,y) pairs
    tex_indices = moc["art_mesh.texture_indices"]

    for pr in packing:
        ob = pr.old_bbox
        if ob.width <= 0 or ob.height <= 0:
            continue

        # New region in UV space [0,1]
        new_min_u = pr.new_x / atlas_size
        new_min_v = pr.new_y / atlas_size
        new_w = pr.new_w / atlas_size
        new_h = pr.new_h / atlas_size

        for mi in ob.mesh_indices:
            vc = vertex_counts[mi]
            if vc == 0:
                continue
            uv_start = uv_begins[mi] * 2  # flat index

            for j in range(vc):
                idx_u = uv_start + j * 2
                idx_v = uv_start + j * 2 + 1

                old_u = uvs[idx_u]
                old_v = uvs[idx_v]

                # Linear transform: old bbox → new bbox
                uvs[idx_u] = (old_u - ob.min_u) / ob.width * new_w + new_min_u
                uvs[idx_v] = (old_v - ob.min_v) / ob.height * new_h + new_min_v

            # Set texture index to 0 (single consolidated sheet)
            tex_indices[mi] = 0
```

    old_textures: dict[int, np.ndarray],
    packing: list[PackedRegion],
    atlas_size: int = 2048,
) -> np.ndarray:
    """Copy pixel regions from old textures to new atlas.

    For each packed region, crops the bounding box from the old texture
    and pastes it into the new atlas at the packed position.
    Resizes if the new region has different pixel dimensions than the old.

    Returns a single RGBA image (atlas_size x atlas_size x 4).
    """
    from PIL import Image

    new_atlas = np.zeros((atlas_size, atlas_size, 4), dtype=np.uint8)

    for pr in packing:
        ob = pr.old_bbox
        if ob.width <= 0 or ob.height <= 0:
            continue

        old_tex = old_textures.get(ob.texture_index)
        if old_tex is None:
            continue

        tex_h, tex_w = old_tex.shape[:2]

        # Source pixel coordinates (clamp to texture bounds)
        src_x0 = max(0, int(np.floor(ob.min_u * tex_w)))
        src_y0 = max(0, int(np.floor(ob.min_v * tex_h)))
        src_x1 = min(tex_w, int(np.ceil(ob.max_u * tex_w)))
        src_y1 = min(tex_h, int(np.ceil(ob.max_v * tex_h)))

        crop = old_tex[src_y0:src_y1, src_x0:src_x1]

        # Resize if needed (region may have different pixel size in new atlas)
        if crop.shape[1] != pr.new_w or crop.shape[0] != pr.new_h:
            pil_crop = Image.fromarray(crop)
            pil_crop = pil_crop.resize((pr.new_w, pr.new_h), Image.LANCZOS)
            crop = np.array(pil_crop)

        # Paste into new atlas
        new_atlas[pr.new_y:pr.new_y + pr.new_h, pr.new_x:pr.new_x + pr.new_w] = crop

    return new_atlas
```

```python
import json


def remap_model(
    moc3_path: Path,
    texture_dir: Path,
    output_dir: Path,
    atlas_size: int = 2048,
    regions: list[RegionDef] | None = None,
) -> dict:
    """End-to-end UV remap: read model, remap UVs, write output.

    Args:
        moc3_path: Path to source .moc3 file
        texture_dir: Directory containing texture_00.png, texture_01.png, etc.
        output_dir: Directory to write modified moc3, new texture, and region_map.json
        atlas_size: Size of new square atlas in pixels
        regions: Custom region definitions (defaults to define_regions())

    Returns:
        Dict with region_map data (same as written to region_map.json)
    """
    from PIL import Image

    if regions is None:
        regions = define_regions()

    # Load moc3
    moc = Moc3.from_file(moc3_path)

    # Load textures
    old_textures: dict[int, np.ndarray] = {}
    for tex_path in sorted(texture_dir.glob("texture_*.png")):
        idx = int(tex_path.stem.split("_")[1])
        old_textures[idx] = np.array(Image.open(tex_path).convert("RGBA"))

    # Compute bboxes and pack
    bboxes = compute_region_bboxes(moc, regions)
    packed = pack_regions(bboxes, atlas_size=atlas_size, padding=4)

    # Remap UVs in moc3
    remap_uvs(moc, packed, atlas_size=atlas_size)

    # Remap textures
    new_atlas = remap_textures(old_textures, packed, atlas_size=atlas_size)

    # Build region map
    region_map = {
        "atlas_size": atlas_size,
        "regions": {},
    }
    for pr in packed:
        region_map["regions"][pr.name] = {
            "x": pr.new_x,
            "y": pr.new_y,
            "w": pr.new_w,
            "h": pr.new_h,
        }

    # Write outputs
    output_dir.mkdir(parents=True, exist_ok=True)
    moc.to_file(output_dir / moc3_path.name)
    Image.fromarray(new_atlas).save(output_dir / "texture_00.png")

    with open(output_dir / "region_map.json", "w") as f:
        json.dump(region_map, f, indent=2)

    # Copy and update model3.json if it exists alongside the moc3
    model3_path = moc3_path.parent / moc3_path.stem.replace(".moc3", "") / ".model3.json"
    # Try common naming patterns
    for candidate in [
        moc3_path.with_suffix(".model3.json"),
        moc3_path.parent / (moc3_path.stem + ".model3.json"),
    ]:
        if candidate.exists():
            model3 = json.loads(candidate.read_text())
            model3["FileReferences"]["Textures"] = ["texture_00.png"]
            with open(output_dir / candidate.name, "w") as f:
                json.dump(model3, f, indent="\t")
            break

    return region_map
```

def draw_region_map(atlas: np.ndarray, region_map: dict) -> np.ndarray:
    """Draw region boundaries and labels on atlas image."""
    vis = atlas.copy()
    if vis.shape[2] == 4:
        # Composite over gray background for visibility
        bg = np.full_like(vis, 128)
        bg[:, :, 3] = 255
        alpha = vis[:, :, 3:4].astype(float) / 255
        vis = (vis[:, :, :3].astype(float) * alpha + bg[:, :, :3].astype(float) * (1 - alpha)).astype(np.uint8)
        vis = np.concatenate([vis, np.full((*vis.shape[:2], 1), 255, dtype=np.uint8)], axis=2)

    colors = [
        (255, 0, 0), (0, 255, 0), (0, 0, 255), (255, 255, 0),
        (255, 0, 255), (0, 255, 255), (128, 0, 0), (0, 128, 0),
        (0, 0, 128), (128, 128, 0), (128, 0, 128), (0, 128, 128),
        (255, 128, 0),
    ]

    for i, (name, r) in enumerate(region_map["regions"].items()):
        color = colors[i % len(colors)]
        x, y, w, h = r["x"], r["y"], r["w"], r["h"]
        cv2.rectangle(vis, (x, y), (x + w, y + h), color, 2)
        cv2.putText(vis, name, (x + 4, y + 20), cv2.FONT_HERSHEY_SIMPLEX, 0.6, color, 2)

    return vis


def main():
    if not HIYORI_DIR.exists():
        print(f"Hiyori not found at {HIYORI_DIR}")
        sys.exit(1)

    OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

    # Run remap
    print("Running UV remap...")
    region_map = remap_model(
        moc3_path=HIYORI_DIR / "hiyori.moc3",
        texture_dir=HIYORI_DIR / "hiyori.2048",
        output_dir=OUTPUT_DIR,

    annotated = draw_region_map(new_atlas, region_map)

    print(f"\nOutputs saved to {OUTPUT_DIR}/")
